# 取消t（打开新页面）映射
unmap t

# 向下滚动半个页面的高度
map J scrollPageDown        

# 向上滚动半个页面的高度
map K scrollPageUp           

# 上一个标签页
map <a-[> previousTab 

# 下一个标签页      
map <a-]> nextTab       

# 新建标签页       
map <a-t> createTab   

# 关闭当前标签页        
map <a-w> removeTab    

# 切换到第一个页面   
map H firstTab     

# 切换最后一个页面                  
map L lastTab                    

 # 在历史记录中后退
map gb goBack        

# 在历史记录中前进          
map gf goForward    

# 切换选择光标
map q focusInput
